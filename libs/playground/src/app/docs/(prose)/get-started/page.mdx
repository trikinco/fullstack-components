# Get started with {NAME_SHORT}

## Installation

```sh
npm install @trikinco/fullstack-components
```

## Setup

1. Create an API handler route to manage the API calls, e.g `app/api/future/route.ts`

2. Import `@trikinco/fullstack-components` and configure the API handler route:

```ts title="app/api/fsc/route.ts"
import {
  handleFSComponents,
  handleNotFoundEnhancement,
  type FSCOptions,
} from '@trikinco/fullstack-components'

const fscOptions: FSCOptions = {
  // Example
  notFoundEnhancer: handleNotFoundEnhancement({
    siteUrl: process.env.SITE_URL || '',
    openAiApiKey: process.env.OPENAI_API_KEY || '',
  }),
  // Additional options and handlers...
}

const fscHandler = handleFSComponents(fscOptions)

export { fscHandler as GET, fscHandler as POST }
```

## Usage

```tsx title="components/NotFoundEnhancment.tsx"
'use client'

import { useNotFoundEnhancement } from '@trikinco/fullstack-components/client'
import Link from 'next/link'

export function NotFoundEnhancer() {
  const { content, isLoading } = useNotFoundEnhancement()

  if (!content || isLoading) {
    return <p>Please wait, checking for other solutions...</p>
  }

  return (
    <div>
      <p>{content?.generatedContent}</p>
      <p>Try this url instead:</p>
      <div>
        <Link href={content?.bestAlternateUrl || '#'}>
          {content?.bestAlternateUrl || 'No alternate url found'}
        </Link>
      </div>
    </div>
  )
}
```
